🔗 Propojení s Google Sheets — sdílené úložiště pro celý tým

Data se vždy ukládají lokálně v prohlížeči. Pro sdílení v týmu (aby viděli všichni) je potřeba jednorázově nasadit Google Apps Script — trvá ~3 minuty:

Krok 1: Jděte na script.google.com → Nový projekt
Krok 2: Smažte veškerý kód a vložte tento:
const SHEET_NAME = 'Inventura2025';

function doGet(e) {
  const id = e.parameter.sheetId;
  const cb = e.parameter.callback;
  const ss = SpreadsheetApp.openById(id);
  let sh = ss.getSheetByName(SHEET_NAME);
  const data = (!sh || sh.getLastRow() < 2) ? [] :
    sh.getRange(2, 1, sh.getLastRow()-1, 7).getValues().map(r => ({
      cislo:r[0], videno:r[1], znamky:r[2], krev:r[3],
      brezost:r[4], cisloVzorku:r[5], poznamka:r[6]
    }));
  const json = JSON.stringify({ok:true, data});
  return ContentService.createTextOutput(cb ? cb+'('+json+')' : json)
    .setMimeType(ContentService.MimeType.JAVASCRIPT);
}

function doPost(e) {
  const d = JSON.parse(e.postData.contents);
  const ss = SpreadsheetApp.openById(d.sheetId);
  let sh = ss.getSheetByName(SHEET_NAME);
  if (!sh) sh = ss.insertSheet(SHEET_NAME);
  sh.clearContents();
  sh.getRange(1,1,1,7).setValues([['cislo','videno','znamky','krev','brezost','cisloVzorku','poznamka']]);
  if (d.rows && d.rows.length)
    sh.getRange(2,1,d.rows.length,7).setValues(d.rows);
  return ok('saved');
}

function ok(data) {
  return ContentService
    .createTextOutput(JSON.stringify({ok:true, data}))
    .setMimeType(ContentService.MimeType.JSON);
}

// Needed for CORS preflight
function doOptions(e) {
  return ContentService.createTextOutput('')
    .setMimeType(ContentService.MimeType.TEXT);
}
Krok 3: Klikněte Nasadit → Nové nasazení → typ: Webová aplikace
→ Spustit jako: Já → Přístup: Kdokoli → Nasadit
Krok 4: Zkopírujte URL nasazení (vypadá jako https://script.google.com/macros/s/ABC.../exec)
Krok 5: Vložte URL níže spolu s ID vašeho Google Sheets
