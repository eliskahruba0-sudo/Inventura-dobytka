const SHEET_NAME = 'Inventura2025'; function doGet(e) { const id = e.parameter.sheetId; const cb = e.parameter.callback; const ss = SpreadsheetApp.openById(id); let sh = ss.getSheetByName(SHEET_NAME); const data = (!sh || sh.getLastRow() < 2) ? [] : sh.getRange(2, 1, sh.getLastRow()-1, 7).getValues().map(r => ({ cislo:r[0], videno:r[1], znamky:r[2], krev:r[3], brezost:r[4], cisloVzorku:r[5], poznamka:r[6] })); const json = JSON.stringify({ok:true, data}); return ContentService.createTextOutput(cb ? cb+'('+json+')' : json) .setMimeType(ContentService.MimeType.JAVASCRIPT); } function doPost(e) { const d = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById(d.sheetId); let sh = ss.getSheetByName(SHEET_NAME); if (!sh) sh = ss.insertSheet(SHEET_NAME); sh.clearContents(); sh.getRange(1,1,1,7).setValues([['cislo','videno','znamky','krev','brezost','cisloVzorku','poznamka']]); if (d.rows && d.rows.length) sh.getRange(2,1,d.rows.length,7).setValues(d.rows); return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); }
